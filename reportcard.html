Python Coding for making a Report Card


grno=int(input("Enter GR.no:"))
name=input("Enter Name:")
#Information on marks in exam
num_sub=int(input("Enter the number of subjects:"))
eng=float(input("Enter English marks:"))
hindi=float(input("Enter Hindi marks:"))
math=float(input("Enter Math marks:"))
sci=float(input("Enter Science Marks:"))
ssc=float(input("Enter Social science marks:"))


total=eng+math+sci+ssc+hindi
average=total/num_sub
percentage=total/100
percentage_1=percentage*100

print("*************************************************")
print(" STUDENT REPORT CARD ")
print("*************************************************")
print("Student Name", name)
print("Student GR.no:", grno)
print("*************************************************")
print("Total marks:", total)
print("*************************************************")
print("Average is:", average)
print("*************************************************")
print("Percentage:",percentage_1)
