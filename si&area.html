Python Coding for Finding Area


# area of square
s=float(input("Enter the size of one side of the square in cm:"))
area=s*s
print("The area of square is:{:.2f}".format (area))

# area of rectangle
l=float(input("Enter the value of the length of the rectangle:"))
b=float(input("Enter the value of the breadth/width of the rectangle:"))
area=l*b
print("{:.2f}".format(area),"is the area of the rectangle.")

#area of circle
r=float(input("Enter the radius of the circle:"))
area=(22/7)*r*r
print("{:.2f}".format(area),"is the area of the circle.")

#area of triangle
b=float(input("Enter the breadth/width of the triangle:"))
h=float(input("Enter the height of the triangle:"))
area=(b*h)/2
print("{:.2f}".format(area),"is the area of the triangle.")
